Variables

Variables are declared with the var keyword: 

var x;
x = 2;

A complete declaration in one line:

var x = 2;

Constants are created with the const keyword. They are unmodifiable.

const color = 'red';

Variables can be strings, numbers, and booleans. These are the "Primitives."

var ball = "Nike";
var goals = 10;
var likesSoccer = true;

Conditional Statements

The if statement is used to execute a block of code if a condition is true.

var myName = 'Fego';
var age = 26;
var isMarried = 'yes';
var lovesToCode = 'yes';

if (age === 26) {
    console.log(myName + ' is ' + age);
}

The else keyword is used to execute a block of code if a condition is false:

if (lovesToCode === 'no') {
    console.log('I am sorry about that');
} else {
    console.log('Way to go ' + myName);
}

The else if keywords introduce other conditions if the first one is false:

if (lovesToCode === 'no') {
    console.log('I am sorry about that');

} else if (isMarried === 'no') {
    console.log('Are you playing videogames too much?');

} else if (age > 35) {
    console.log('Are you kidding me?');

} else {
    console.log('Pheww... you are ok');
}

The switch statement provides a way to introduce conditionals. The break keyword "breaks" the statement so that other cases are not evaluated unnecessarily:

switch ('OLA') {
    case 'HELLO':
        console.log('This is English');
        break; // this breaks the switch statement so that
               // the other cases are not evaluated unnecessarily.

    case 'HOLA': // different cases can share the same code block
    case 'OLA':
        console.log('This is Spanish (HOLA) and Portuguese (OLA),
        they almost sound the same.');
        break;

    case 'BONJOUR':
        console.log('This is French');
        break;

    case 'NAMASTE':
        console.log('This is Hindi');

    default: // default expression in case the main expression
             // is not found
        console.log('Who knows what this is');
}

Functions

Functions in JavaScript are declared with the function keyword and may return values with the return statement:

function calculateNumberOfMonths(fromYear, toYear) {

    var result;
    result = ((toYear - fromYear) * 12);
    return result;
}

The result of a function can be assigned to a variable:

var numberOfMonths = calculateNumberOfMonths(1990, 2018);
console.log(numberOfMonths);

Or not be assigned:

console.log('This is the number of months from 1970: '
   + calculateNumberOfMonths(1970, 2018) + ' months.');

Function expressions are meant to produce a value:

// this is a function expression

var numberOfYears = function(fromYear, toYear) {
    return toYear - fromYear;
}
console.log('Using a function expression: '
+ numberOfYears(1980, 2020));

If a variable is declared outside of the function it is considered global and can be accessed by any function:

var x = 10;
function(){
    console.log(x);
}

if a variable is declared without the var keyword inside the function, it is considered global too.

function myName() {
    var firstname = "Fernando";
    secondName = "Gomez";
    console.log(firstname + ' ' + secondName);
}

console.log(secondName);

Function Parameters

Function parameters are the names or characters we see inside the () in a function. These parameters are also called arguments and are passed-by-value meaning that if the function changes the value of the parameter, the orginal value will still be preserved:


function bestYears(x, y, z){ // A function can an unlimited number of
                             // parameters
    console.log(x);
}

Loops

For loops provide a way to execute a specific functionality until a condition is met: 


for (var i = 0; i < 5; i++) {
    console.log(i); // prints all values from 0 to 24
}

Statement 1 can be omitted. Just make sure that values are initialized before the loop: 


var g = 0;
for (; g < 10; g++) {
    console.log(g);
}

Statement 2 can be omitted too, but a "break" must be inside the loop to end the loop: 


for (var h = 5;; h--) {
    if (h === 0) {
        break;
    }
    console.log(h);
}

Statement 3 can be omitted as well:


for (var k = 7; k > 0;) {
    console.log(k);
    k--;
}

While loops can provide the same functionality. As long as the condition inside () is true, then code block will keep looping:


console.log('While Loop');
var j = 0; // initialize variable.
while (j < 10) {
    console.log(j);
    j++; // incrementor. This is necessary
         // otherwise loop will never end.

The do/while loop executes code block once, before checking if the condition is true, then it will repeat the loop as long as the condition is true.


var result = ""
var b = 0;
do {
    console.log('do-while-loop!')
    b++;
}
while (b < 5);
